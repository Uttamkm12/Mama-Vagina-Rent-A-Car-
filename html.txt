    <script>
        // Telegram WebApp Initialization
        let tg = window.Telegram.WebApp;
        tg.expand();

        let incomeTotal = 0;
        let outgoingTotal = 0;

        function saveData() {
            const type = document.getElementById('type').value;
            const amount = parseFloat(document.getElementById('amount').value);
            const note = document.getElementById('note').value;

            if(!amount || amount <= 0) { alert("সঠিক পরিমাণ দিন"); return; }

            if(type === 'income') incomeTotal += amount;
            else outgoingTotal += amount;

            updateUI();
            
            // এখানে পরে আমরা Google Sheets এ ডাটা পাঠানোর কোড যোগ করব
            alert("ডাটা সেভ হয়েছে!");
            document.getElementById('amount').value = '';
            document.getElementById('note').value = '';
        }

        function updateUI() {
            document.getElementById('totalIncome').innerText = incomeTotal;
            document.getElementById('totalOutgoing').innerText = outgoingTotal;
            const history = document.getElementById('history');
            history.innerHTML = "সর্বশেষ এন্ট্রি যোগ করা হয়েছে।";
        }
    </script>
